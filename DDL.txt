CREATE DATABASE SALEPOINT

CREATE TABLE Store (
	ID INT IDENTITY (1, 1) PRIMARY KEY,
	StoreName VARCHAR (255) NOT NULL,
	Phone VARCHAR (25),
	Email VARCHAR (255),
	City VARCHAR (255)
);


CREATE TABLE Brand (
	ID INT IDENTITY (1, 1) PRIMARY KEY,
	BrandName VARCHAR (255) NOT NULL
);
